git init - инициализация репозитория
git status - проверка информации по ветке, коммитам и файлам
git add название.файла - добавление файла для отслеживания
git add . - отслеживает все новые изменения файла, кроме удаления файлов
git add -A - отслеживает новые изменения файла + удаленные файлы
git commit -m "название коммита" - вносим коммиты
git config --global user.name - задаю имя
git config --global user.email - задаю емейл 
git log --oneline - показывает все изменения
git remote add origin https://github.com/nvrndng/TopPython54.git - подключаю папку к репозиторию на гитхабе
git push -u origin main - тоже подключаю папку к репозиторию
